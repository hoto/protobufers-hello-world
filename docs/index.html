PROTOBUF

Serializes structured data for use in communication protocols.


                    message Person {
                      required int32 id = 1;
                      required string name = 2;
                      optional string email = 3;
                    }

Generates library code for multiple languages.


                    Person andy = Person.newBuilder()
                        .setId(123456)
                        .setName("Andrzej Rehmann")
                        .setEmail("andrzej.rehmann@gmail.com")
                        .build();
                    output = new FileOutputStream(args[0]);
                    andy.writeTo(output);
